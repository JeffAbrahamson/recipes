\dish{Carrot Raisin Peanut Salad}
\serves{4}
\prep{15 minutes}
\source{joy}

\begin{ingredients}
  \ingr{4}{}{large carrots (10 oz.)}
  \ingr{\fracH}{c}{raisins}
  \ingr{\fracH}{c}{coarsely chopped peanuts}
  \ingr{\fracQQ}{tsp}{salt}
  \ingr{}{}{freshly ground black pepper}
  \ingr{2}{Tbsp}{grated lemon peel}
  \ingr{1}{Tbsp}{lemon juice}
  \ingr{\fracH}{c}{cultured sour cream, more to taste}
  \ingr{\fracH}{c}{mayonnaise, more to taste}
  \ingr{}{dash}{cayenne pepper}
\end{ingredients}


\begin{recipe}
  \begin{enumerate}
  \item Scrape carrots.  Place on ice for an hour, then grate coursely into a bowl.

  \item Add remaining ingredients except sour cream and mayonnaise.  Mix lightly.

  \item Add sour cream and mayonnaise.

  \end{enumerate}
\end{recipe}
