\dish{Salad 102}
\altdish{Egglplant and Tomato Salad}
\altdish{Tomato and Egglplant Salad}
\altdish{Salade \`a l'aubergine et \`a la tomate}
\altdish{Salade \`a la tomate et \`a l'aubergine}
\serves{4}
%\makes{}
\prep{}
\source{stephane.birkle}

\textbf{Salad body:}

\begin{ingredients}
  \ingr{1}{}{egglant (approx 275 g)}
  \ingr{2}{}{red bell peppers (approx 500 g)}
  \ingr{250}{g}{yellow cherry tomato}
  \ingr{\fracQ}{}{red onion, minced}
  \ingr{1}{}{baby cucumber (approx 150 g), finely sliced}
  \ingr{80}{g}{quinoa}
  \ingr{3}{\tbsp}{fresh coriander leaf, chopped}
  \ingr{3}{\tbsp}{fresh basil, chopped}
\end{ingredients}

\textbf{Dressing:}

\begin{ingredients}
  \ingr{1}{\tsp}{mustard}
  \ingr{\fracH}{\tsp}{salt}
  \ingr{\fracH}{\tsp}{black pepper}
  \ingr{\fracH}{\tsp}{ground cumin}
  \ingr{\fracH}{\tsp}{ground coriander}
  \ingr{1}{\tsp}{honey}
  \ingr{1}{\tbsp}{balsamic vinegar}
  \ingr{1}{\tbsp}{olive oil}
  \ingr{1}{\tbsp}{water}
  \ingr{}{}{}
\end{ingredients}


\begin{recipe}
  \begin{enumerate}

  \item Halve and roast the bell peppers.  Remove the skin, and cut in
    thin strips.

  \item Bake the eggplant until it is soft.  Remove the skin and cut
    in thin strips.

  \item Mix the salad body.

  \item Mix the salad dressing.

  \item Combine just before serving.

  \end{enumerate}

  \textbf{Variations:}
  \begin{itemize}
  \item Add chick peas
  \item Add mozzarella balls
  \item Use lime juice instead of balsamic vinegar
  \item Use Tabasco instead of black pepper
  \item Use Thai basil
  \end{itemize}
\end{recipe}
