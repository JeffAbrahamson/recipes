\dish{Taboulé quinoa betterave}
\altdish{Quinoa betterave taboulé}
\altdish{betterave quinoa taboulé}
\serves{4}
%\makes{}
\prep{20 minutes}
\source{atelier:papillon}

\stage{Salade}

\begin{ingredients}
  \ingr{100}{g}{quinoa}
  \ingr{2}{}{betteraves crues}
  \ingr{1}{}{céleri branche}
  \ingr{1}{}{botte persil ou de la menthe (selon goût)}
  \ingr{1}{}{oignon nouveau}
  \ingr{1}{}{poignée canneberges séchées}
\end{ingredients}

\stage{Vinaigraitte}

\begin{ingredients}
  \ingr{3}{\cs}{huile d'olive}
  \ingr{3}{\cs}{vinaigre balsamique}
  \ingr{\fracH}{\ccf}{cumin}
  \ingr{}{}{sel}
  \ingr{}{}{poivre}
\end{ingredients}


\begin{recipe}
  \begin{enumerate}

  \item Faire cuire le quinoa dans une grande casserole d'eau
    bouillante pendant 15~minutes avec une pincée de sel.

  \item Découper en petits morceaux les betteraves, feuilles de céleri
    branche et oignon nouveau.  Ciseler les feuilles de menthes et/ou
    de persil.

  \item Égoutter le quinoa et laisse refroidir.

  \item Réaliser la vinaigrette.

  \item Mélanger le tout puis ajouter les canneberges.  Laisser
    refroidir au réfrigérateur avant de servir.

  \end{enumerate}
\end{recipe}
