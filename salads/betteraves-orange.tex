\dish{Salade de Betteraves \`a l'Orange}
\altdish{beet orange salad}
\altdish{orange beet salad}
\altdish{betteraves, salade \`a l'orange}
\altdish{orange, salade de betteraves}
\serves{}
\makes{}
\prep{15 minutes + 1 heure de repos}
\source{cuisineAZ}

\begin{ingredients}
  \ingr{300}{g}{betterave cuite}
  \ingr{1}{}{orange}
  \ingr{10}{brins}{persil}
  \ingr{1}{\ccf}{grain de cumin}
  \ingr{1}{\ccf}{moutarde forte}
  \ingr{1}{\cs}{vinaigre de vin}
  \ingr{3}{\cs}{huile d'olive}
  \ingr{}{}{sel}
  \ingr{}{}{poivre}
\end{ingredients}


\begin{recipe}
  \begin{enumerate}

  \item Pelez les betteraves et coupez-les en fines tranches.

  \item Pressez la moiti\'e d'une orange et versez le jus dans un
    saladier avec le vinaigre ; faites-y fondre le sel. Ajoutez la
    moutarde et 4 tours de moulin Ã  poivre ; m\'elangez en ajoutant
    l'huile d'olive et le cumin.

  \item Faites mac\'erer les tranches de betterave 1~h dans cette
    sauce.

  \item Avant de servir, pelez la seconde moiti\'e d'orange \`a vif et
    d\'ecorez avec les demis quartiers d\'ebarrass\'es de leur
    membrane.

  \item Lavez le persil et ciselez-en quelques feuilles au dessus de
    la salade.

  \end{enumerate}
\end{recipe}
