\dish{Couscous Courgette}
\altdish{Courgette couscous}
\serves{2--4}
%\makes{}
\prep{30 minutes}
\source{stephane.birkle}

\begin{ingredients}
  \ingr{\fracH}{c}{couscous}
  \ingrS{}{}{raisins secs}

  \ingr{}{}{beurre}
  \ingr{1}{}{oignon}
  \ingr{2}{}{courgettes (jaunes), coupées en ronds}
  \ingr{}{}{fêta}
  \ingr{}{}{menthe}
\end{ingredients}


\begin{recipe}
  \begin{enumerate}

  \item Faire cuire le couscous dans de l'eau bouillante.  À la fin de
    la cuisson, ajoutée des raisins secs.

  \item Faire revenir l'oignon dans du beurre.  Puis ajouté les
    courgettes.  À la fin de la cuisson, ajoutée la fêta et la
    menthe.

  \item Servir chaud, les légumes sur un lit de couscous.

  \end{enumerate}
\end{recipe}
