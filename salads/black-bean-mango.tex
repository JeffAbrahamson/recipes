\dish{Black Bean Salad with Mango}
\altdish{Mango and Black Bean Salad}
\serves{4}
%\makes{}
\prep{}
\source{unknown}


\textbf{Black beans:}

\begin{ingredients}
  \ingr{1}{lb}{black beans}
  \ingr{1}{}{small onion}
  \ingr{2}{}{ribs celery}
  \ingr{2}{}{carrots, cut in two-inch pieces}
  \ingr{\fracH}{}{green bell pepper}
  \ingr{3}{}{cloves garlic}
  \ingr{1}{}{bay leaf}
  \ingr{3}{}{sprigs fresh thyme (or 1 tsp dry)}
  \ingr{3}{}{sprigs parsley}
  \ingr{\fracH}{tsp}{ground cumin}
  \ingr{\fracH}{tsp}{dried oregano}
  \ingr{}{}{salt}
  \ingr{}{}{pepper}
\end{ingredients}


\begin{recipe}
  \begin{enumerate}

  \item Soak beans in cold water to cover by at least 3 inches for at
    least 4 hours.

  \item Put bay leafe, thyme, and parsley in a cheese cloth or tea ball.

  \item Add all ingredients except salt and pepper.  Bring to a boil
    over high heat.  Skim off any foam.

  \item Reduce heat and gently simmer uncovered, stirring
    occasinoally, until tender, 60-75 minutes.  Add water as necessary
    to keep the beans submerged.

  \item Season with salt and pepper during the last ten minutes of
    cooking.

  \item Drain and rinse with cold water.  Remove and discard the
    vegetables and the spice bag.

  \end{enumerate}
\end{recipe}


\textbf{Marinade:}

\begin{ingredients}
  \ingr{1}{Tbsp}{olive oil}
  \ingr{1}{Tbsp}{lime juice}
  \ingr{\fracH}{tsp}{ground coriander}
  \ingr{}{}{salt}
  \ingr{}{}{pepper}
\end{ingredients}


\begin{recipe}
  \begin{enumerate}

  \item The marinade is for shrimp that I don't use.  The marinade
    portion of this recipe may be obsolete here.

  \end{enumerate}
\end{recipe}


\textbf{Salad:}

\begin{ingredients}
  \ingr{1}{}{mango, ripe}
  \ingr{2}{c}{black beans, above}
  \ingr{4}{Tbsp}{chopped fresh mint leaves (or 1 Tbsp dry)}
  \ingr{4}{}{scallions, trimmed and minced}
  \ingr{2}{Tbsp}{olive oil}
  \ingr{2}{Tbsp}{lime juice}
  \ingr{}{}{salt}
  \ingr{}{}{pepper}
\end{ingredients}


\begin{recipe}
  \begin{enumerate}

  \item Peel and seed the mango.  Dice into \fracH-inch cubes.

  \item Immediately before serving, combine the mango, black beans, 3
    Tbsp of the mint leaves, scallions, olive oil, lime juice, salt,
    and pepper and toss.  The salad should be highly seasoned.

  \end{enumerate}
\end{recipe}
