\dish{Hot and Sour Soup}
\serves{8}
\prep{30 minutes}
\source{msm}

\begin{ingredients}
  \ingr{1}{}{large onion, chopped}
  \ingr{1}{qt}{broth or water}
  \ingr{20}{oz}{mushrooms (crimini or even button)}
  \ingr{1--2}{}{zucchini, shredded}
  \ingr{4}{Tbsp}{soy sauce}
  \ingr{2}{Tbsp}{sherry}
  \ingr{\fracQ}{c}{red wine or cider vinegar}
  \ingr{}{}{Tabasco sauce}
  \ingr{\fracH}{tsp}{black pepper}
  \ingr{12}{oz}{tofu, shredded}
  \ingr{2}{}{eggs, lightly beaten}
  \ingr{}{}{scallions, minced}
  \ingr{}{}{sesame oil}
\end{ingredients}


\begin{recipe}
  \begin{enumerate}

  \item Saut\'e onions until translucent.  Add water, mushrooms, and
  zucchini.  Simmer until mushrooms are done.

  \item Add soy sauce, sherry, wine, Tabasco, pepper, and finally
  tofu.  Simmer about 10 minutes.

\item Add eggs in a thin stream while stirring.

\item Top with scallions and dots of sesame oil to serve.

  \end{enumerate}
\end{recipe}
