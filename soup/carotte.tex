\dish{Soupe de carotte aux \'epices}
\altdish{Spicy carrot soup}
\altdish{Carrot soup, spicy}
\altdish{Carotte, soupe aux \'epices}
\pourD{4}{en entr\'ee}
%\makes{}
\prep{30 minutes + 30 minutes cuisson}
\source{unknown}

\begin{ingredients}
  \ingr{2}{\cs}{huile de tournesol}
  \ingr{1}{}{oignon hach\'e}
  \ingr{3}{gousses}{d'ail, \'ecras\'ees}
  \ingr{2}{\ccf}{cumin, en poudre}
  \ingr{1}{\ccf}{coriandre, en poudre}
  \ingr{\fracH}{\ccf}{gingembre, en poudre}
  \ingr{1}{\ccf}{paprika}
  \ingr{1}{pinc\'ee}{piment de Cayenne}
  \ingr{1}{}{pomme de terre, petite, coup\'ee en cubes}
  \ingr{500}{g}{carottes, coup\'ees en rondelles}
  \ingr{1,2}{l}{bouillon de l\'egumes ou de poule}
  \ingr{}{}{jus d'une orange}
  \ingr{1--1\fracH}{\cs}{vinaigre de vin rouge}
  \ingr{}{}{sel}
  \ingr{}{}{poivre noir}
  \ingr{}{}{coriandre fra\^iche, cisel\'ee}
\end{ingredients}


\begin{recipe}
  \begin{enumerate}

  \item Faire chauffer l'huile dans une grande casserole.  Y mettre
    l'ail et l'oignon \`a revenir 4~minutes sur feu doux.

  \item Ajouter le cumin, la coriandre, le gingembre, le paprika, et
    le piment de Cayenne.

  \item Ajouter la pomme de terre, les carottes, et le bouillon.

  \item Porter le tout \`a \'ebullition, puis r\'eduire la
    temp\'erature et couvrir.  Laisser mijoter 20~minutes environ :
    les l\'egumes doivent \^etre tendres.

  \item Passer la soupe au mixeur jusqu'\`a l'obtention d'un
    velout\'e.

  \item Incorporer le jus d'orange et le vinaigre.  Saler et pouvrer
    au go\^ut.

  \item Transvaser la soupe \`a la louche dans des bols et parsemer de
    coriandre fra\^iche.  Servir aussit\^ot.

  \end{enumerate}
\end{recipe}
