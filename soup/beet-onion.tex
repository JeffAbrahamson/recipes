\dish{Beet Onion Soup}
\altdish{Onion Beet Soup}
\serves{6}
%\makes{}
\prep{1 hour}
\source{jma}

\begin{ingredients}
  \ingr{4}{}{medium onions, finely chopped}
  \ingr{50}{g}{butter}
  \ingr{4}{}{garlic cloves, crushed}
  \ingr{4}{}{raw beets, finely chopped}
  \ingr{5}{}{tomatoes, coarsely chopped}
  \ingr{}{}{salt}
  \ingr{}{}{black pepper}
  \ingr{2}{c}{water}
  \ingr{4}{}{yellow bell peppers}
  \ingr{}{}{handful fresh basil}
  \ingr{1}{tsp}{thyme}
  \ingr{}{}{cr\`eme fra\^iche}
\end{ingredients}


\begin{recipe}
  \begin{enumerate}

  \item Saut\'e onions slowly in butter, 15 minutes.

  \item Add garlic and beets, saut\'e another 10 minutes.

  \item Add tomatoes, salt, and pepper.  Simmer on low heat 15 minutes.

  \item Add water and bell pepper.  Cook on low 5--10 minutes.

  \item Stir in the basil and thyme.

  \item Before serving, stir in several large spoonfuls of cr\`eme fra\^iche.

  \end{enumerate}
\end{recipe}
