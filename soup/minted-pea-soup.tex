\dish{Minted Pea Soup}
\altdish{Pea Soup with Mint}
\serves{4--6}
%\makes{}
\prep{20 minutes plus 30 minutes to cook}
\source{lagasse2001pea,lagasse1993new-orleans}

\begin{ingredients}
  \ingr{2}{Tbsp}{butter, unsalted}
  \ingr{1}{c}{yellow onion, chopped}
  \ingr{1}{tsp}{essence, recipe below}
  \ingr{\fracH}{tsp}{salt}
  \ingr{1}{qt}{vegetable stock}
  \ingr{4}{c}{peas, freshly shelled}
  \ingr{\fracH}{c}{heavy cream}
  \ingr{\fracQ}{c}{fresh chopped mint}
\end{ingredients}


\begin{recipe}
  \begin{enumerate}

  \item In a large, heavy saucepan, melt the butter over medium-high
    heat. Add the onion, essence, and salt, and saute until soft,
    about 5 minutes.

  \item Add the chicken stock and peas and bring to a boil. Reduce the
    heat to medium-low and simmer uncovered until the peas are tender,
    about 5 to 10 minutes. Remove from the heat and puree with a
    hand-held immersion blender.

  \item Stir in the cream and mint, and serve warm with bread.

  \end{enumerate}
\end{recipe}

\textbf{Essence (Creole seasoning)}


\begin{ingredients}
  \ingr{2\fracH}{Tbsp}{paprika}
  \ingr{2}{Tbsp}{salt}
  \ingr{2}{Tbsp}{garlic powder}
  \ingr{1}{Tbsp}{black pepper}
  \ingr{1}{Tbsp}{onion powder}
  \ingr{1}{Tbsp}{cayenne pepper}
  \ingr{1}{Tbsp}{dried leaf oregano}
  \ingr{1}{Tbsp}{dried thyme}
\end{ingredients}


\begin{recipe}
  \begin{enumerate}

  \item Combine all ingredients thoroughly and store in an airtight
    container.  Makes about 2/3 cup.

  \end{enumerate}
\end{recipe}


The recipe originally called for chicken stock.
