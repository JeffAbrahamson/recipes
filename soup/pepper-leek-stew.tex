\dish{Pepper, Leek, and Mushroom Stew with Corn and Cheese Dumplings}
\altdish{Leek, Pepper, and Mushroom Stew with Corn and Cheese Dumplings}
\altdish{Mushroom, Leek, and Pepper Stew with Corn and Cheese Dumplings}
\altdish{Corn and Cheese Dumplings, for Leek, Pepper, and Mushroom Stew}
\altdish{Cheese and Corn Dumplings, for Leek, Pepper, and Mushroom Stew}
\serves{4}
%\makes{}
\prep{30--40 minutes (dumplings) + 30 minutes (stew)}
\source{still:life}

\stage{Corn and Cheese Dumplings}

\begin{ingredients}
  \ingr{100}{g}{grated cheddar cheese (originally: \fracH cup)}
  \ingr{100}{g}{grated Swiss cheese (originally: \fracH cup)}
  \ingr{\fracH}{c}{yellow corn meal}
  \ingr{\fracH}{c}{white flour}
  \ingr{\fracH}{c}{cottage cheese}
  \ingr{2}{}{eggs, beaten}
  \ingr{1}{pinch}{salt}
\end{ingredients}


\begin{recipe}
  \begin{enumerate}

  \item Mix all ingredients together thoroughly in a bowl.  Use hands
    to form 1-inch balls.

  \item In a large pot, bring 4 quarts of water to a rolling boil.
    Add as many dumplings as possible without crowding.  Reduce heat,
    cover, and allow to simmer for 10 minutes.  Remove with a slotted
    spoon.  (\textit{Note}: If cooking the dumplings in more than one
    shift, keep cooked ones warm on a tray in a 200\F\ oven or reheat
    in a microwave.)

  \end{enumerate}
\end{recipe}

The dumplings may be assembled a day or two in advance if desired.
Store on a floured plate, tightly covered and refrigerated.  About 40
minutes before serving time, bring the cooking water to a boil.  Poach
the dumplings while heating the stew.


\stage{Pepper, Leek, and Mushroom Stew}

\begin{ingredients}
  \ingr{2}{Tbsp}{butter}
  \ingr{2\fracH}{c}{chopped leaks (whites and half the greens)(not
    quite one leek)}
  \ingr{\fracH}{lb}{button mushrooms, chopped}
  \ingr{\fracH}{tsp}{salt}
  \ingr{6}{}{bell peppers, thinly sliced}
  \ingr{}{}{black pepper, to taste}
  \ingr{1}{c}{yogurt, at room temperature}
  \ingr{}{}{paprika}
\end{ingredients}

\begin{recipe}
  \begin{enumerate}

  \item Melt the butter in a large saucepan or Dutch oven.  Cook the
    vegetables with salt over low heat.  After about 10 minutes, turn
    up the heat and cook, stirring until most of the liquid
    evaporates.  Add black pepper to taste.

  \item Stir in the yogurt shortly before serving.  Serve immediately
    without any further cooking.  Top each serving with a few of the
    dumplings and an artful dusting of paprika.
    
  \end{enumerate}

  Notes:
  \begin{itemize}
  \item The vegetables may be cooked a day or two in advance and then
    reheated.

  \item Be sure the yogurt is at room temperature so the sauce won't
    curdle.

  \end{itemize}
\end{recipe}
