\dish{Carrot Soup with Onion Relish}
\altdish{Onion Relish for Carrot Soup}
\serves{4--6}
%\makes{}
\prep{1 hour}
\source{madison.VCfE}

\stage{Soup}

\begin{ingredients}
  \ingr{2}{Tbsp}{butter or olive oil}
  \ingr{1}{}{onion, thinly sliced}
  \ingr{1}{lb}{carrots, thinly sliced}
  \ingr{1}{bay}{leaf}
  \ingr{2}{Tbsp}{chopped parsley}
  \ingr{3}{Tbsp}{white rice}
  \ingr{1}{tsp}{sweet paprika}
  \ingr{1}{tsp}{ground cumin}
  \ingr{\fracH}{tsp}{ground coriander}
  \ingr{}{}{salt and black pepper}
  \ingr{7}{c}{water or stock}
\end{ingredients}


\begin{recipe}
  \begin{enumerate}

  \item In a soup pot, melt the butter over medium heat.  Add the
    onion, carrots, bay leaf, parsley, and rice.  Cook to soften the
    onion, stirring frequently, about 5 minutes.

  \item Add the spices, \fracH tsp salt, some pepper, and cook 5
    minutes longer.

  \item Add the water and bring to a boil, then lower the heat and
    simmer, partially covered, for 25 minutes.

  \item While the soup is cooking, make the relish, below.

  \item Remove the bay leaf from the soup.  Puree 2 cups of the soup
    until smooth, then puree the rest, leaving a little texture and
    flecks of carrot.  Taste for salt and serve each bowl with a
    spoonful of the relish.
  \end{enumerate}
\end{recipe}


\stage{Onion Relish}

\begin{ingredients}
  \ingr{\fracH}{c}{finely diced white onion or pickled onion}
  \ingr{1}{}{serrano chile, finely chopped}
  \ingr{2}{Tbsp}{chopped cilantro}
  \ingr{}{}{a few leaves of cinnamon basil (or ordinary basil)}
  \ingr{2}{}{limes (grated zest and juice)}
\end{ingredients}


\begin{recipe}
  \begin{enumerate}

  \item Combine all ingredients together.

  \end{enumerate}
\end{recipe}
