\dish{Spinach Mancino}
%\altdish{}
\serves{4--6}
%\makes{}
\prep{30 minutes}
\source{unknown}

\begin{ingredients}
  \ingr{1}{}{onion, chopped}
  \ingr{}{}{olive oil}
  \ingr{}{}{garlic, crushed}
  \ingr{1}{lb}{fresh spinach}
  \ingr{}{}{raisins}
  \ingr{}{}{slivered almonds}
  \ingr{}{}{salt}
  \ingr{}{}{pepper}
\end{ingredients}


\begin{recipe}
  \begin{enumerate}

  \item Cook onion in olive oil.  Add garlic.  Add spinach and let
    wilt, cooking to remove excess moisture.  Add raisins and slivered
    almonds.  Season.  Serve.

  \end{enumerate}
\end{recipe}
