\dish{Tuna Mousse}
\altdish{Mousse, tuna}
\serves{4--8}
%\makes{}
\prep{20 minutes}
\source{flemming.behrendt}

\begin{ingredients}
  \ingr{3$\times$280}{g}{canned tuna (840~g tuna), finely ground}
  \ingr{5}{}{biscotti}
  \ingr{4}{}{eggs}
  \ingr{150}{g}{butter}
  \ingr{2}{\tbsp}{flour}
  \ingr{1}{}{garlic clove}
  \ingr{}{}{fresh parsley}
  \ingr{}{}{juice of one lemon}
\end{ingredients}


\begin{recipe}
  \begin{enumerate}

  \item Combine the ingredients.  Place in a buttered bread pan and
    bake 35~minutes at 180\C.

  \end{enumerate}
\end{recipe}
