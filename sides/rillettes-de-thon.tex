\dish{Rillette de thon}
\altdish{Thon, rillette}
%\serves{}
%\makes{}
\prep{10 minutes}
\source{celine-martin}

\begin{ingredients}
  \ingr{150}{g}{fromage frais (type St Môret}
  \ingr{1}{boite}{thon au naturel}
  \ingr{1}{}{échalote (petite)}
  \ingr{\fracH}{}{jus de citron vert}
  \ingr{\fracH}{}{boite de ciboulette}
  \ingr{}{}{baies roses}
  \ingr{}{}{poivre}
\end{ingredients}


\begin{recipe}
  \begin{enumerate}

  \item Mélanger le tout en hachant.  Conserver au frais au moins 3~heures.

  \end{enumerate}
\end{recipe}
