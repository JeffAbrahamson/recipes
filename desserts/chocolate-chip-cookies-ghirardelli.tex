\dish{Ghirardelli Bittersweet Chocolate Chip Cookies}
\altdish{Chocolate Chip Cookies, Ghirardelli Bittersweet}
\altdish{Cookies, Chocolate Chip, Ghirardelli Bittersweet}
%\serves{}
\makes{36}
\prep{15 minutes + 10 minutes to bake}
\source{ghirardelli-insert}

\begin{ingredients}
  \ingr{\fracH}{c}{(115~g) butter}
  \ingr{\fracH}{c}{(110~g) sugar}
  \ingr{\fracH}{c}{(100~g) packed brown sugar}
  \ingr{1}{}{egg}
  \ingr{1}{tsp}{vanilla}
  \ingr{1}{c + 2 Tbsp}{flour}
  \ingr{\fracH}{tsp}{baking soda}
  \ingr{\fracH}{tsp}{salt}
  \ingr{8}{oz}{(225~g) bittersweet chocolate, chipped}
\end{ingredients}


\begin{recipe}
  \begin{enumerate}

  \item Cream together butter, sugars, egg, and vanilla.

  \item Stir together flour, baking soda, and salt.  Blend into butter
    mixture.

  \item Mix in chipped chocolate.

  \item Drop by rounded teaspoon onto ungreased baking sheets.  Bake
    at 375\F\ (190\degreeC) for 8--10 minutes, until light golden brown.  Cool on
    racks.

  \end{enumerate}
\end{recipe}
