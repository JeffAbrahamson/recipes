\dish{Chocolate Chip Raisin Cookies}
\altdish{Cookies, chocolate chip raisin}
\makes{50 cookies}
\prep{45 minutes}
\source{anne:abrahamson}        % not 100% sure that this is her recipe

\begin{ingredients}
  \ingr{2 \fracQ}{c}{flour}
  \ingr{1}{tsp}{baking soda}
  \ingr{1}{tsp}{salt}
  \ingr{1}{c}{butter}
  \ingr{\fracQQ}{c}{white sugar}
  \ingr{\fracQQ}{c}{brown sugar}
  \ingr{1}{tsp}{vanilla}
  \ingr{2}{}{eggs}
  \ingr{2}{c}{raisons}
  \ingr{12}{oz}{chocolate chips (2 c)}
\end{ingredients}


\begin{recipe}
  \begin{enumerate}
  \item Beat butter well.  Add sugars and beat until creamy.  Add vanilla.

  \item Mix flour, salt, and baking soda together and set aside.

  \item Add eggs to butter and sugars.  Add flour mixtrue a little at
  a time, beating well after each addition.

\item Add chocolate chips, then raisins.

\item Drop onto ungreased cookie sheets and bake 8--10 minutes at 375
  F.   Remove from sheets immediately.

  \end{enumerate}
\end{recipe}
