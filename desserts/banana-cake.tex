\dish{Banana Cake with Cinnamon and Chocolate}
\altdish{Cinnamon and Chocolate Banana Cake}
\altdish{Chocolate and Cinnamon Banana Cake}
\altdish{Cake, Banana with Chocolate and Cinnamon}
%\serves{}
\makes{one $9\times 13$ pan}
\prep{30 minutes + 1 hour to bake}
\source{jma}

\begin{ingredients}
  \ingr{330}{g}{T55 flour, sifted (2 \fracQ c plus 5 Tbsp)}
  \ingr{10}{g}{baking soda (1 \fracH tsp)}
  \ingr{5}{g}{salt (\fracH tsp)}
  \ingr{230}{g}{yogurt (1 cup)}
  \ingr{5}{}{very ripe bananas (skins should be brown)}
  \ingr{230}{g}{unsalted butter}
  \ingr{300}{g}{brown sugar}
  \ingr{1}{tsp}{vanilla extract}
  \ingrS{4}{}{large eggs, separated}
  \ingrS{100}{g}{chocolate chips}
  \ingr{}{}{cinnamon and sugar (50/50 mixture)}
\end{ingredients}


\begin{recipe}
  \begin{enumerate}

  \item Preheat oven to \CF{175}{350}.  Lightly butter a $13\times 9$ inch
    baking pan.

  \item Sift flour, baking soda, and salt together into a small bowl
    and set aside.

  \item In a second small bowl, mash bananas, then stir in the yogurt.

  \item Cream butter, sugar, and vanilla until light and fluffy,
    about 2 minutes.  Add the eggs one at a time and beat on medium
    until blended, about 10 seconds each.

  \item Stir in one third of the dry ingredients, then half the banana
    mixture, then the remainder of the dry, then the remainder of the
    banana.  Stir briefly after each addition.

  \item Pour batter into the prepared pan.  Cover with a thick layer
    of cinnamon and sugar.

  \item Bake at \CF{175}{350} until the cake springs back to the touch and a
    tester inserted in the center comes out dry, about 1 hour 10
    minutes.

  \item Cool on a wire rack.  When cool, frost with chocolate ganache (below).

  \end{enumerate}
\end{recipe}


\textbf{Ganache:}

\begin{ingredients}
  \ingr{80}{g}{chocolate Kumuntu (80\%)}
  \ingr{80}{g}{chocolate Guanaja (70\%)}
  \ingr{160}{g}{cream}
  \ingr{}{}{cocoa powder}
\end{ingredients}


\begin{recipe}
  \begin{enumerate}

  \item Gently melt butter and chocolate.  Mix in cream.

  \item Frost the cake.  Dust with cocoa.  Allow to rest at least an
    hour before serving.

  \end{enumerate}
\end{recipe}
