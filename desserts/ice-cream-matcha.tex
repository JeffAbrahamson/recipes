\dish{Glace au thé matcha}
\altdish{Matcha ice cream}
\altdish{Ice cream, matcha}
%\serves{}
\fait{1 litre}
\prep{20 minutes plus congélation}
\source{jma}

\stage{Base}

\begin{ingredients}
  \ingr{25}{g}{matcha}
  \ingrS{230}{ml}{eau}
  \ingr{150}{g}{sugar}
  \ingr{pince}{}{sel}
  \ingrS{450}{g}{crème}
  \ingrS{3}{}{jaune d'oeufs}
  \ingrS{2}{\tbsp}{vodka}
  \ingr{60}{g}{meringue italienne~(page \pageref{label:meringue-italienne-gelato})}
\end{ingredients}


\begin{recipe}
  \begin{enumerate}

  \item Infuser le matcha à 85\degreeC.
    
  \item Chauffer la crème, sucre, et sel.  Ajouter lentement aux oeufs
    en fouettant.  Remettre sur feu doux est faire une crème anglaise
    (80\degreeC).

  \item Verser le thé matcha dans la crème.  Laisser refroidir, puis
    passer par une tamise afin d'enlever les feuilles de thé.  Laisser
    refroidir à 5\degreeC.
    
  \item Ajouter le vodka et turbiner, ajoutant la meringue au moment
    que la préparation commence à épaissir.

  \end{enumerate}
\end{recipe}
