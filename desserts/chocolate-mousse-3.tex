\dish{Variations sur une mousse au chocolate}
\altdish{Chocolate mousse, variations}
\altdish{Mousse, chocolate, variations}
\prep{15 minutes plus plusieurs heures de repos}
\source{vincent.guerlais}

\begin{ingredients}
  \ingr{240}{g}{cr\`eme liquide}
  \ingr{130}{g}{chocolate, 70\%}
\end{ingredients}


\begin{recipe}
  \begin{enumerate}

  \item M\'elanger la cr\`eme en chantilly, puis ajouter le chocolat
    \`a 40\C.  Laisser refroider plusieurs heures.

  \end{enumerate}

\end{recipe}


Variation de Jeff :

\begin{ingredients}
  \ingr{240}{g}{cr\`eme liquide}
  \ingr{50}{g}{cacao en poudre}
\end{ingredients}


\begin{recipe}
  \begin{enumerate}

  \item M\'elanger la cr\`eme en chantilly, puis ajouter le cacao.  Le
    m\'elange devient tr\`es \'epais.  Peut servir tout de suite ou
    rester au refrigerateur.

  \end{enumerate}

  \textit{La quantit\'e de cacao n'est pas s\^ur.}
  
\end{recipe}
