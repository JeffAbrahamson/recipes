\dish{Glace au chocolat blanc}
\altdish{White chocolate ice cream}
\altdish{Ice Cream, white chocolate}
%\serves{}
\fait{1 litre}
\prep{20 minutes plus congélation}
\source{jma}

\stage{Base}

\begin{ingredients}
  \ingr{180}{g}{chocolat blanc}
  \ingrS{50}{g}{sugar}
  \ingr{300}{ml}{lait}
  \ingr{pince}{}{sel}
  \ingrS{3}{}{jaune d'oeufs}
  \ingrS{200}{ml}{crème}
  \ingrS{2}{\tbsp}{vodka}
  \ingr{60}{g}{meringue italienne~(page \pageref{label:meringue-italienne-gelato})}
\end{ingredients}


\begin{recipe}
  \begin{enumerate}

  \item Faire fondre ou ramollir le chocolat blanc.
    
  \item Chauffer le lait, sucre, et sel.  Ajouter lentement aux oeufs
    en fouettant.  Remettre sur feu doux est faire une crème anglaise
    (80\degreeC).

  \item Tamiser sur le chocolat blanc.  Remuer, puis ajouter la crème.
    Laisser refroidir à 5\degreeC.
    
  \item Ajouter le vodka et turbiner, ajoutant la meringue au moment
    que la préparation commence à épaissir.

  \end{enumerate}
\end{recipe}
