\dish{Sorbet au chocolat}
\altdish{Chocolat, sorbet}
\altdish{Gelato, chocolat}
%\serves{}
\fait{400 ml}
\prep{20 minutes + temps de macération + congélation}
\source{pierre.herme.10}

\begin{ingredients}
  \ingr{320}{ml}{eau (originelle : minérale)}
  \ingr{50}{g}{sucre semoule}
  \ingr{40}{g}{glucose (originelle : sucre inverti)}
  \ingr{15}{g}{cacao poudre}
  \ingr{70}{g}{chocolat (originelle : Valrhona Madagascar)}
  \ingrS{40}{g}{meringue italienne~(page \pageref{label:meringue-italienne-gelato})}
\end{ingredients}

(La recette originelle propose 3~g de SL~66 stabilisateur de sorbet.
La recette originelle ne propose pas de meringue.)


\begin{recipe}
  \begin{enumerate}

  \item Faire fondre le chocolat, y dissoudre le cacao poudre.

  \item Chauffer l'eau, verser sur les autres ingrédients (sauf le
    meringue).  Mixer.  Laisser refroidir 24~heures au réfrigérateur.

  \item Mixer et mettre dans une sorbetière et congeler.

  \item Vers la fin de la congelation, ajouter le meringue.

  \end{enumerate}
\end{recipe}
