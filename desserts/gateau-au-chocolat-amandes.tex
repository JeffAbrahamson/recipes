\dish{Gateau au chocolat aux amandes}
\altdish{Chocolate almond cake}
\altdish{Cake, chocolate almond}
\altdish{Flourless chocolate almond cake}
\altdish{Almond chocolate cake}
\serves{16}
\prep{30 minutes + 1 heure de cuisson}
\source{stephane.birkle}

Une recette de Naples.

\begin{ingredients}
  \ingr{300}{g}{chocolat}
  \ingr{300}{g}{d'amandes émond\'ees concass\'ees}
  \ingr{300}{g}{sucre}
  \ingr{200}{g}{beurre}
  \ingr{6}{}{oeufs}
\end{ingredients}

\begin{recipe}
  \begin{enumerate}

  \item Battre les blanc en neige.

  \item M\'elanger les jaunes au sucre, les faire blanchir au batteur
    électrique.  Ajouter le beurre mou puis le chocolat fondu.

  \item Incorporer d\'elicatement les blancs en neige.

  \item Faire cuire \`a 350\F pour 50 minutes ou jusqu'\`a ce que le
    gateau soit solide au milieu.  (Un teste de couteau ne marche pas
    bien avec ce gateau.)

  \end{enumerate}
\end{recipe}
