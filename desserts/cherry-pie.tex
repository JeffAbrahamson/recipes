\dish{Cherry Pie}
\altdish{Pie, cherry}
%\serves{}
\makes{1 pie}
\prep{15 minutes plus various resting and baking times}
\source{joy}

\begin{ingredients}
  \ingr{4}{c}{fresh sour cherries}
  \ingr{2\fracTT}{\tbsp}{quick-cooking tapioca}
  \ingr{1\fracT}{c}{sugar}
  \ingr{2}{\tbsp}{kirsch (or 2 drops almond flavoring)}
  \ingr{2}{\tbsp}{butter}
\end{ingredients}


\begin{recipe}
  \begin{enumerate}

  \item Prepare a double crust pie crust, page~\pageref{pie-crust}.

  \item Wash, drain, and pit the cherries.

  \item Combine the tapioca, sugar, and kirsch, then gently mix with
    the cherries.  Let the mixture stand 15~minutes.

  \item Preheat oven to \FC{450}{232}.

  \item Pour fruit into pie dough and dot with butter.  Cover with the
    upper crust, and bake 10~minutes at \FC{450}{232}.

  \item Reduce heat to \FC{350}{177} and bake about 40~minutes longer
    or until golden brown.

  \item Allow to cool thoroughly before slicing, as the cherries will
    not gel properly when still hot.
    
  \end{enumerate}
\end{recipe}
