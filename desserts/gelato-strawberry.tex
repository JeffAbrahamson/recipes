\dish{Strawberry Gelato}
\altdish{Gelato, Strawberry}
\serves{4--8}
%\makes{}
\prep{20 minutes (plus time for chilling)}
\source{unknown}

\begin{ingredients}
  \ingr{1}{c}{(less 1 Tbsp) sugar}
  \ingr{1}{c}{water}
  \ingr{450}{g}{strawberries}
  \ingr{1}{\tbsp}{lemon juice}
  \ingr{1}{\tsp}{egg white}
\end{ingredients}


\begin{recipe}
  \begin{enumerate}

  \item Make a syrup of the water and sugar.  Chill.

  \item Trim strawberries, pur\'ee until smooth.  Stir in the syrup
    and egg white.  Chill.

  \item Mix in a sorbeti\`ere.

  \end{enumerate}
\end{recipe}
