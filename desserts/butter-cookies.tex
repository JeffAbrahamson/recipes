\dish{Butter Cookies}
\altdish{Cookies, butter}
%\serves{\ }
\makes{130 cookies}
\prep{45 minutes}
\source{anne:abrahamson}

\begin{ingredients}
  \ingr{1}{lb}{butter}
  \ingr{1 \fracH}{c}{sugar}
  \ingr{4}{}{eggs}
  \ingr{2}{tsp}{vanilla}
  \ingr{1}{tsp}{salt}
  \ingr{8}{c}{flour}
  \ingr{2}{tsp}{baking powder}
\end{ingredients}


\begin{recipe}
  \begin{enumerate}
  \item Cream butter well.  Add sugar and beat until smooth.
    Add eggs, one at a time.  Add vanilla.  Add salt, baking powder,
    and flour.  Beat well.

  \item Form cookies.

  \item Bake 10--12 minutes at 375--400\F\ (190--205\degreeC).

  \end{enumerate}
\end{recipe}
