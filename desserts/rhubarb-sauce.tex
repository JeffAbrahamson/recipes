\dish{Rhubarb Sauce}
%\altdish{}
%\serves{}
\makes{2 cups}
\prep{10 minutes + 30 minutes to bake}
\source{jim.gladstone}

\begin{ingredients}
  \ingr{500}{g}{rhubarb}
  \ingr{\fracH}{c}{sugar}
  \ingr{}{}{balsamic vinegar}
\end{ingredients}


\begin{recipe}
  \begin{enumerate}

  \item Preheat oven to at 400\F (200\degreeC).

  \item Cut the rhubarb into 1 cm pieces.  Place in a baking pan,
    sprinkle with sugar, drizzle some balsamic vinegar on top.

  \item Bake for 30 minutes.

  \end{enumerate}
\end{recipe}
