\dish{Tarte Metzinger}
\aussidish{Streussel aux cerises}
\altdish{Cerises, streussel}
\altdish{Cherry streussel}
\altdish{Streusel, cherry}
\pour{10}
\prep{5 hours}
\sourcep{à base des recettes de Jean Millet \cite{jean.millet} et
  inspiré d'un gâteau servi en mai 2014 au café Reichard à Cologne.
}

\stage{Pâte à brioche}

\begin{ingredients}
  \ingr{65}{ml}{lait}
  \ingr{5}{g}{levure}
  \ingrS{25}{g}{farine}
  \ingr{100}{g}{farine}
  \ingr{15}{g}{sucre}
  \ingr{2,5}{g}{sel}
  \ingr{1}{}{oeuf}
  \ingr{50}{g}{beurre à température ambiante}
\end{ingredients}


\begin{recipe}
  \begin{enumerate}

  \item Tamiser la farine.

  \item Mélanger le lait, la levure, et les 25~g de farine.  Ajouter
    une pincée de sucre.  Laisser former une bouille pendant
    30-45~minutes.

  \item Ajouter les ingrédients restants et pétrir le tout jusqu'à ce
    que la pâte soit lisse.

  \item Couvrir et laisser lever de 2 à 3~heures.

  \end{enumerate}
\end{recipe}

\vbox{
\stage{Crème pâtissière}

\begin{ingredients}
  \ingr{500}{ml}{lait}
  \ingrS{2}{gousses}{vanille}
  \ingr{5}{}{jaunes d'oeufs}
  \ingrS{100}{g}{sucre}
  \ingr{30}{g}{farine}
  \ingr{1}{pincée}{sel}
\end{ingredients}
}

\begin{recipe}
  \begin{enumerate}

  \item Fendre la gousse de vanille dans la longueur, gratter
    l'intérieur et ajouter au lait.  Porter à ébullition.

  \item Travailler les jaunes d'oeufs et le sucre en mousse, puis
    incorporer la farine.

  \item Ajouter le lait buillant, remettre le tout dans une casserole
    à feu doux sans cesser de fouetter.  Donner un bouillon.

  \item Passer la crème à travers un chinois ou un tamis.  Laisser
    refroidir.  {\it Saupoudrer de sucre glace pour éviter la formation
    d'une peau.  Ou laisser refroidir dans le mixer à vitesse basse
    avant de la passer à travers le tamis.}

  \end{enumerate}
\end{recipe}

\stage{Streussel}

\begin{ingredients}
  \ingr{250}{g}{farine}
  \ingr{150}{g}{beurre}
  \ingr{150}{g}{sucre}
  \ingr{1}{}{jaune d'oeuf}
  \ingr{1}{cuillerée}{cannelle}
\end{ingredients}


\begin{recipe}
  \begin{enumerate}

  \item Mélanger le tout pour obtenir des grumeaux.

  \end{enumerate}
\end{recipe}

\stage{Gâteau}

\begin{ingredients}
  \ingr{500}{g}{cérises dénoyautées}
\end{ingredients}

\begin{recipe}
  \begin{enumerate}

  \item Chauffer le four à 180\degreeC.

  \item Pétrir la pâte.

  \item Beurrer un moule de 25~cm.

  \item Étaler la pâte à 1~cm d'épaisseur dans le moule.

  \item Étaler la crème sur la pâte.  Laisser reposer 30~minutes.

  \item Ranger les cerises sur la crème.

  \item Parsemer le streussel.

  \item Faire cuire à 180\degreeC{} pendant{} 40~minutes.

  \end{enumerate}
\end{recipe}
