\dish{Oatmeal Raisin Cookies}
\altdish{Cookies, oatmeal raisin}
\makes{about 72 cookies}
\prep{30 minutes}
\source{pathmark:oatmeal}

\begin{ingredients}
  \ingr{1}{c}{butter}
  \ingr{1}{c}{sugar}
  \ingr{1}{c}{packed brown sugar}
  \ingr{2}{}{eggs}
  \ingr{1}{tsp}{vanilla}
  \ingr{2}{c}{flour}
  \ingr{2}{tsp}{baking powder}
  \ingr{\fracQQ}{tsp}{cinnamon}
  \ingr{\fracH}{tsp}{salt}
  \ingr{2}{c}{oats}
  \ingr{3}{c}{raisins}
\end{ingredients}


\begin{recipe}
  \begin{enumerate}
  \item Preheat oven to 350 $^\circ$F.

  \item Cream butter and sugar.  Mix in eggs and vanilla.

  \item Sift together flour, baking powder, cinnamon, and salt.  Add
    to creamed mixture.  Blend thoroughly.
    
  \item Stir in oats, then raisins.

  \item Drop by level tablespoons 2 inches apart onto baking sheets.
  Bake 11--13 minutes.

\item Cool 1 minute before removing from baking sheets.

  \end{enumerate}
\end{recipe}
