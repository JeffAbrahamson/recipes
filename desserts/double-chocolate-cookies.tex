\dish{Ghirardelli Double Chocolate Chip Cookies}
\altdish{Ghirardelli Double Chocolate Chip Cookies}
\altdish{Chocolate chip cookies, Ghirardelli}
\altdish{Cookies, chocolate chip}
\makes{4 dozen}
\prep{30 minutes}
\source{ghirardelli}

\begin{ingredients}
  \ingr{$2\times 6$}{oz}{(170~g~$\times 2$) semi-sweet chocolate chips}
  \ingr{\fracH}{c}{(115~g) butter}
  \ingrS{1}{}{egg}
  \ingr{110}{g}{sugar (\fracH c)}
  \ingr{75}{g}{packed brown sugar (\fracH c)}
  \ingr{1}{tsp}{vanilla}
  \ingrS{1 \fracTT}{c}{flour}
  \ingr{\fracH}{tsp}{baking soda}
  \ingr{\fracH}{tsp}{baking powder}
  \ingr{\fracH}{tsp}{salt}
  \ingrS{\fracQ}{c}{milk}
  \ingr{1}{tsp}{lemon juice}
\end{ingredients}


\begin{recipe}
  \begin{enumerate}
  \item Melt 1 cup chocolate chips with butter.

  \item Beat egg with sugar, brown sugar and vanilla until creamed.
    Mix in melted chocolate.

  \item Stir flour with baking soda, baking powder and salt.

  \item Combine milk with lemon juice, stirring until thick.

  \item Add dry ingredients to chocolate mixture alternately with
    milk.  Stir in remaining cup chocolate chips.  Chill dough.

  \item Drop by teaspoon onto ungreased baking sheet.  Bake at
    \CF{175}{350} for 8~to 10~minutes.  Cool on rack.
    
  \end{enumerate}

  Suggestions: Try beating sugar into softened butter, then adding
  egg.  Melt chocolate without exceeding 50\degreeC{}.
\end{recipe}
