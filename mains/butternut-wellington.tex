\dish{Butternut Wellington}
\altdish{Wellington de butternut}
\altdish{Butternut en croûte}
\serves{6}
%\makes{}
\prep{35~minutes préparation + 45~minutes cuisson}
\source{cvf-web}

\begin{ingredients}
  \ingr{850}{g}{butternut, coupé en dés}
  \ingr{400}{g}{échalotes, hâchées}
  \ingrS{2}{\cs}{huile d'olive}
  \ingr{50}{g}{noix de pécan, hâchés}
  \ingr{1--2}{\cs}{sirop d'érable}
  \ingrS{1\fracH}{\cs}{vinaigre balsamique}
  \ingrS{500}{g}{\hyperref[pâte-feuilletée]{pâte feuilletée}}
  \ingr{1}{\cs}{sauge}
  \ingrS{200}{g}{fromage bleu, coupé en dés}
  \ingr{1}{}{oeuf, battu}
\end{ingredients}


\begin{recipe}
  \begin{enumerate}

  \item Chauffer le four à 200\degreeC.

  \item Mélanger butternut et échalotes avec l'huile, assaisonner,
    faire cuire 20~minutes.

  \item Ajouter les noix de pécans et continuer jusqu'à ce que le
    butternut soit cuit mais encore ferme.

  \item Ajouter le sirop d'érable et le vinaigre balsamique.

  \item Laisser refroidir.

  \item Étaler la pâte, placer la préparation dessus, puis parsemer de
    sauge et de bleu.

  \item Fermer avec l'oeuf, puis dorer le tout.  Faire quelques petits trous.

  \item Enfourner 30--35~minutes.

  \end{enumerate}
\end{recipe}
