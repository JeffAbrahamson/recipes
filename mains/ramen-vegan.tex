\dish{Ramen vegan aux trois champignons}
\altdish{精進ラーメン}
\hiragana{しょうじんラーメン}
\pour{4}
\prep{20~min + cuisson (20~min)}
\source{cjm}


\stage{Soupe}

\begin{ingredients}
  \ingr{2}{\cs}{huile végétale neutre} \ingr{1}{gousse}{ail, haché}
  \ingr{100}{g}{champignons enokis} \ingr{60}{g}{pleurotes}
  \ingrS{10}{g}{shiitaké séché, réhydraté dans le bouillon vegan}
  \ingr{}{}{\hyperref[bouillon-legumes]{bouillon vegan pour ramen}}
  \ingr{3}{\cs}{mirin} \ingr{3}{\cs}{levure maltée} \ingr{}{}{sel}
\end{ingredients}

\begin{recipe}
  \begin{enumerate}

  \item Effilocher les enokis, séparer les pleurotes, couper le
    shiitaké en quatre.

  \item Chauffer l'huile et l'ail~; dès que ça embaume, ajouter les
    champignons (1~min). Verser le bouillon et les assaisonnements,
    porter à ébullition, couvrir et mijoter 10~min. Saler.

  \end{enumerate}
\end{recipe}


\stage{Huile parfumée}

\textit{葱油 (ねぎあぶら)}

\begin{ingredients}
  \ingr{2}{}{cébettes, en très fines rondelles}
  \ingr{5}{g}{gingembre frais, pelé et émincé}
  \ingr{1}{gousse}{ail, haché}
  \ingrS{1}{}{grand piment rouge (piment d'Espelette ou piment chinois),
    en rondelles de 1~cm}
  \ingr{150--200}{ml}{huile de tournesol ou de sésame}
\end{ingredients}

\begin{recipe}
  \begin{enumerate}

  \item Couvrir les ingrédients d'huile juste à hauteur dans une petite
    casserole. Chauffer jusqu'aux premières bulles, baisser à feu doux
    et cuire 3~min.

  \item Filtrer quand les cébettes sont dorées. Réserver cébettes et
    piment frits (condiment).

  \end{enumerate}
\end{recipe}


\stage{Composition}

\begin{ingredients}
  \ingr{320}{g}{\hyperref[ラーメン]{nouilles ramen sèches ou maison}}
  \ingrS{2}{}{choux bok choy, coupés en quatre dans la longueur}
  \ingr{}{}{poivre}
\end{ingredients}

\begin{recipe}
  \begin{enumerate}

  \item Cuire les nouilles~; ajouter les bok choy 1~min avant la fin.
    Égoutter.

  \item Servir dans 4~bols avec la soupe et les bok choy. Arroser de
    1--2~\ccf\ de negiabura. Poivrer.

  \end{enumerate}
\end{recipe}
