\dish{Curry de lotte}
\altdish{Lotte au curry}
\serves{4}
%\makes{}
\prep{25 minutes}
\source{internet}

\begin{ingredients}
  \ingr{800}{g}{lotte, coupé en morceaux}
  \ingr{1}{}{poivron jaune, petit}
  \ingr{1}{}{poivron route, petit}
  \ingr{1}{}{oignon, gros}
  \ingr{200}{ml}{lait de coco}
  \ingr{1}{\cs}{pâte de curry (ou en poudre)}
  \ingr{}{}{sel}
  \ingr{}{}{poivre}
\end{ingredients}


\begin{recipe}
  \begin{enumerate}

  \item Émincer l'oignon, couper les poivrons en petits cubes.  Faire
    revenir dans de l'huile.

  \item Ajouter la pâte de curry, la lotte, et le lait de coco.
    Laisser mijoter 10~minutes sur feu doux.

  \item Saler et pouvrer.  Servir.

  \end{enumerate}
\end{recipe}
