\dish{Chickpea Eggplant Curry}
\altdish{Eggplant Chickpea Curry}
\altdish{Curry, eggplant chickpea}
\serves{4--6}
%\makes{}
\prep{75 minutes}
\source{chris.card}

\begin{ingredients}
  \ingr{1}{\tsp}{salt}
  \ingr{\fracQQ}{\tsp}{ground coriander}
  \ingr{\fracQQ}{\tsp}{ground cumin}
  \ingr{\fracH}{\tsp}{turmeric}
  \ingr{\fracH}{\tsp}{red pepper flakes (optional)}
  \ingr{\fracQ}{\tsp}{cinnamon}
  \ingr{\fracQ}{\tsp}{black pepper}
  \ingrS{\fracQ}{\tsp}{ground cardamom}
  \ingrS{1}{}{eggplant, large, diced (in France, probably two eggplants)}
  \ingr{2}{\tbsp}{ghee}
  \ingr{3}{cloves}{garlic, minced}
  \ingrS{3}{cm}{ginger, grated or minced}
  \ingrS{1}{}{onion, finely chopped}
  \ingrS{125}{ml}{tomato paste (\fracQ c)}
  \ingrS{}{}{chick peas (200 g dry, cooked and drained)}
  \ingrS{550}{ml}{broth or water (2 cups)}
  \ingr{}{}{coriander leaf}
\end{ingredients}


\begin{recipe}
  \begin{enumerate}

  \item Combine all spices, mix and set aside.

  \item Heat a 30~cm non-stick skillet over medium heat.  Add half of
    the diced eggplant and cook dry for 7~minutes.  Set aside and
    repeat with the other half.  Add a tablespoon of water or two if
    the pan sticks.

  \item Using the same skillet, heat the ghee, then add the garlic and
    ginger.  Cook over medium-high heat for about 2~minutes.

  \item Add the onion and spice mix, cook until translucent,
    5--6~minutes.

  \item Add the tomato paste and cook 3--4~minutes, stirring
    frequently.

  \item Add the chick peas, stir and cook for 2--3~minutes.

  \item Add the cooked eggplant and the broth or water.  Bring to a
    boil.  Cover, reduce heat and simmer 40~minutes.

  \item Add the cilantro during the final 10~minutes of cooking.

  \end{enumerate}
\end{recipe}
