\dish{Strudel aux épinards}
\altdish{Épinards, strudel}
\altdish{Spinach strudel}
\serves{6}
%\makes{}
\prep{45 minutes + 30 minutes + cuisson + pâte feilletée}
\source{jma}

\begin{ingredients}
  \ingr{700}{g}{oignons finement hâchés}
  \ingr{}{}{beurre salé}
  \ingrS{}{}{sel}
  \ingr{350}{g}{épinards hâchés (poids des feuilles, vers le double en botte)}
  \ingr{}{}{poivre}
  \ingr{}{}{poivre de jamaïque}
  \ingrS{2}{}{oeufs}
  \ingr{200}{g}{fêta}
  \ingr{}{}{parmesan râpé}
  \ingrS{1}{tasse}{chapelure}
  \ingr{500}{g}{\hyperref[pâte-feuilletée]{pâte feuilletée}}
\end{ingredients}


\begin{recipe}
  \begin{enumerate}

  \item Faire revenir les oignons dans du beurre.
    
  \item Ajouter les épinards, laissé couvert une minute, puis faire
    cuire jusqu'à ce que la liquide soit évaporée.  Épicer.

  \item Laisser refroidir.  Lorsque tiède, ajouter oeufs, fromages, et
    chapelure.

  \item Rouler la pâte feuilletée.  Diviser en deux.  Placer la moitié
    de la farce sur chaque pâte vers un côté, plier les deux côtés
    vers l'intérieur, et rouler pour former deux strudels.
    
  \item Faire cuire à 200\degreeC{} en chaleur tournante penant 30~minutes.

  \end{enumerate}
\end{recipe}
