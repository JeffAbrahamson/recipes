\dish{Yetakelt W'et}
\altdish{W'et yetakelt}
\altdish{Ethiopian: yetakelt w'et}
\serves{6}
\prep{20 minutes + 40 minutes to cook}
\source{unknown}

\label{yetakelt-wet}

\begin{ingredients}
  \ingr{1}{c}{onion}
  \ingr{2}{cloves}{garlic}
  \ingr{1}{Tbsp}{berbere}
  \ingr{1}{Tbsp}{Hungarian paprika}
  \ingr{\fracQ}{c}{niter kebbeh, page~\pageref{niter-kebbeh}}
  \ingr{1}{c}{green beans, cut into thirds}
  \ingr{1}{c}{carrots}
  \ingr{1}{c}{potato, cubed}
  \ingr{1}{c}{tomato, chopped}
  \ingr{\fracQ}{c}{tomato paste}
  \ingr{2}{c}{vegetable stock}
  \ingr{}{}{salt}
  \ingr{}{}{black pepper}
  \ingr{\fracQ}{c}{parsley, fresh, chopped}
\end{ingredients}


\begin{recipe}
  \begin{enumerate}

  \item Saut\'e the onions, garlic, berbere, and paprika in the niter
    kebbeh for 2 minutes.  Add the beans, carrots, and potatoes and
    continue to saut\'e for about 10 minutes, stirring occasionally to
    prevent burning.

  \item Add the chopped tomatoes, tomato paste, and the vegetable
    stock.  Bring to a boil and then simmer for 15 minutes or until
    all of the vegetables are tender.

  \item Add salt, pepper.  Mix in the parsley.

  \item Serve on injera.

  \end{enumerate}

Good served with Yetakelt W'et, page~\pageref{yemiser-wet}.

\end{recipe}
