\dish{Vegetable Stroganoff}
\altdish{Stroganoff vegetable}
\serves{6}
\prep{1 hour}
\source{moosewood}


\textbf{The sauce:}

\begin{ingredients}
  \ingr{1}{c}{chopped onion}
  \ingr{\fracH}{lb}{mushroom, chopped}
  \ingr{2}{Tbsp}{butter}
  \ingr{3}{c}{sour cream}
  \ingr{1\fracH}{c}{yogurt}
  \ingr{3}{Tbsp}{dry red wine}
  \ingr{\fracQQ}{tsp}{salt}
  \ingr{\fracQ}{tsp}{dill weed}
  \ingr{dash}{}{tamari sauce}
  \ingr{}{}{paprika}
  \ingr{}{}{black pepper}
\end{ingredients}


\begin{recipe}
  \begin{enumerate}

  \item Saut\'e onions and mushrooms in butter until onions are soft.

  \item Combine all ingredients in the top of a double boiler and heat
    genly about 30 minutes.

  \end{enumerate}
\end{recipe}


\textbf{Vegetables and Noodles:}

\begin{ingredients}
  \ingr{6}{c}{broccoli, cauliflower, carrots, zucchini, celery,
    cabbage, peppers, cherry tomatoes, etc.}
  \ingr{4}{c}{egg noodles}
\end{ingredients}

If making fresh pasta, 2 eggs plus 300 g flour is about right.


\begin{recipe}
  \begin{enumerate}

  \item Steam 6 cups of chopped fresh vegetables.

  \item Cook 4 cups of raw egg noodles in boiling, salted water until
    tender.  Drain and butter.

  \item Assemble the stroganoff on a platter and garnish with
    freshly-minced scallions.

  \end{enumerate}
\end{recipe}
