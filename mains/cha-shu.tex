\dish{叉焼}
\hiragana{チャーシュー}
\altdish{Cha-shū}
\altdish{Ajitsuke tamago}
\altdish{味付け卵}
\altdish{Bouillon ramen}
\pour{4}
\prep{30~min + repos (2~h à 1~nuit) + cuisson (2~h~30)}
\source{cjm}
\label{chashu}


\stage{Dashi concentré}

\begin{ingredients}
  \ingr{500}{ml}{eau}
  \ingr{5}{g}{kombu séché}
  \ingr{10}{g}{katsuobushi (bonite séchée)}
\end{ingredients}

\begin{recipe}
  \begin{enumerate}

  \item Porter à ébullition, frémir 10~min. Laisser refroidir,
    égoutter en pressant le katsuobushi.

  \end{enumerate}
\end{recipe}


\stage{Marinade du cha-shū}

\begin{ingredients}
  \ingr{150}{ml}{sauce soja}
  \ingr{2}{\cs}{saké}
  \ingr{1}{\cs}{mirin}
  \ingr{1}{\cs}{sucre non raffiné}
  \ingr{1}{tranche ($5 \times 5$~cm)}{kombu séché}
\end{ingredients}

\begin{recipe}
  \begin{enumerate}

  \item Réunir tous les ingrédients dans un grand sachet à zip.
    Réserver.

  \end{enumerate}
\end{recipe}


\stage{Ajitsuke tamago}

\textit{味付け卵 (あじつけたまご)}

\begin{ingredients}
  \ingr{4}{}{oeufs, à température ambiante}
\end{ingredients}

\begin{recipe}
  \begin{enumerate}

  \item Cuire 6~min~30~secondes. Refroidir à l'eau froide, écaler, tremper
    1~min dans l'eau. Égoutter.

  \end{enumerate}
\end{recipe}


\stage{Bouillon à la viande}

\begin{ingredients}
  \ingr{500}{g}{rôti d'épaule de porc, ficelé}
  \ingrS{4}{}{ailes de poulet fermier}
  \ingr{2}{L}{eau}
  \ingr{\fracH}{}{oignon, coupé en deux}
  \ingr{\fracH}{}{carotte, en rondelles de 2~cm}
  \ingr{1}{}{shiitaké séché}
  \ingr{1}{}{vert de poireau}
  \ingr{100}{ml}{saké}
  \ingr{2}{tranches (5~mm)}{gingembre frais, avec la peau}
  \ingr{2}{gousses}{ail}
\end{ingredients}

\begin{recipe}
  \begin{enumerate}

  \item Blanchir les ailes et le porc 1~min. Égoutter.

  \item Porter tous les ingrédients à ébullition dans une grande
    casserole. Baisser à feu doux, couvrir partiellement et mijoter
    1~h~30.

  \item Au bout d'1~heure, retirer le rôti et laisser refroidir.
    Ajouter le dashi concentré et mijoter à découvert encore 45~min.
    Filtrer. Environ 1,6~L de bouillon.

  \item Ajouter le porc et les oeufs dans le sachet de marinade, retirer
    l'air et fermer. Mariner au moins 2~h (idéalement une nuit).

  \end{enumerate}
\end{recipe}
