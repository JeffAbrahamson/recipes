\dish{Tartes aux ognions}
\altdish{Ognions, tarte}
\altdish{Onion tarte}
\serves{6--8}
%\makes{}
\prep{30 min + 30 minutes to bake}
\source{stephane.birkle}

\begin{ingredients}
  \ingr{1}{}{p\^ate \`a pain}
  \ingr{1}{kg}{ognions}
  \ingr{1}{}{oeuf}
  \ingr{50}{g}{beurre}
  \ingr{2}{cuill\`ere \`a soupe }{cr\`eme fraiche}
  \ingr{}{}{muscade}
  \ingr{2}{cuill\`ere \`a soupe}{farine}
  \ingr{}{}{sel}
\end{ingredients}


\begin{recipe}
  \begin{enumerate}

  \item Faire une p\^ate \`a pain.

  \item Pr\'echauffer le four \`a 210\C (410\F).

  \item Couper les ognions en direction de l'axe de l'ognion.
    Faire revenir dans le beurre.
    M\'elanger le tout sauf la p\^ate \`a pain.

  \item \'Etaler la p\^ate \`a pain.  Mettre le m\'elange
    l\`a-dessus.  Faire cuire 30 minutes ou jusqu'\`a ce que le pain
    soit cuit et les ognions dor\'ees.

  \end{enumerate}
\end{recipe}
