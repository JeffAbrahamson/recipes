\dish{Pasta in Vodka Sauce}
\altdish{Vodka Sauce on Pasta}
\serves{4(?)}
\prep{30 min ?}
\source{dave.richardson}

\begin{ingredients}
  \ingr{1}{lb}{pasta}
  \ingr{10}{cloves}{garlic}
  \ingr{6}{Tbsp}{vodka}
  \ingr{2}{28 oz cans}{tomatoes}
  \ingr{2}{Tbsp}{butter}
  \ingr{2}{Tbsp}{basil}
  \ingr{}{}{salt}
  \ingr{}{}{black pepper}
  \ingr{1\fracH}{c}{cream}
  \ingr{20}{}{TVP meatballs, cooked 7 minutes}
\end{ingredients}


\begin{recipe}
  \begin{enumerate}

  \item Chop garlic and saut\'ee in butter.

  \item Add tomatoes, basil, salt, pepper, and vodka.  Cook until thick and chunky.

  \item Add cream.  Cook another 5 minutes.

  \item Serve over pasta with TVP meatballs.

  \end{enumerate}
\end{recipe}
