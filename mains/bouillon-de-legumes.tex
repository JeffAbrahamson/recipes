\dish{Bouillon de légumes pour ramen}
\altdish{Bouillon ramen vegan}
\altdish{Ramen, bouillon de légumes}
\altdish{ラーメン, bouillon de légumes}
\pour{4}
\prep{10~min + repos (1~nuit) + cuisson (45~min)}
\source{cjm}
\label{bouillon-legumes}

\begin{ingredients}
  \ingr{10}{g}{kombu séché}
  \ingr{10}{g}{shiitaké séché}
  \ingrS{1}{L}{eau (pour l'infusion)}
  \ingr{1,4}{L}{eau}
  \ingr{160}{g}{chutes de légumes (épluchures de carotte, tige de
    brocoli, trognon de chou, feuilles de chou chinois, pieds de
    champignon\ldots)}
  \ingr{1}{petit}{oignon, coupé en quatre}
  \ingr{1}{}{vert de poireau}
  \ingr{5}{g}{gingembre}
  \ingr{2}{gousses}{ail}
\end{ingredients}


\begin{recipe}
  \begin{enumerate}

  \item Infuser le kombu et le shiitaké dans 1~L d'eau une nuit (ou
    2~h dans de l'eau chaude).

  \item Porter tous les ingrédients à ébullition, couvrir partiellement
    et mijoter 40~min.

  \item Filtrer.

  \end{enumerate}
\end{recipe}
