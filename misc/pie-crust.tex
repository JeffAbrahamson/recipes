\dish{Pie Crust}
\altdish{Crust for fruit pie}
%\serves{}
\makes{1 pie crust}
\prep{15 minutes plus refrigerator time}
\source{rosie}

\label{pie-crust}

\textbf{Single Crust for a 9 Inch Pie}

\begin{ingredients}
  \ingr{1\fracH}{c}{all purpose flour}
  \ingr{\fracQ}{\tsp}{salt}
  \ingr{9}{\tbsp}{butter (128 g)}
  \ingr{3}{\tbsp}{cold water}
\end{ingredients}


\vspace{5mm}
\textbf{Double Crust for a 9 Inch Pie}

\begin{ingredients}
  \ingr{2\fracQ}{c}{all purpose flour}
  \ingr{\fracH}{\tsp}{salt}
  \ingr{13\fracH}{\tbsp}{butter (192 g)}
  \ingr{4\fracH}{\tbsp}{cold water}
\end{ingredients}


\begin{recipe}
  \textbf{Food Processor Version}

  \begin{enumerate}

  \item Process flour and salt for 20 seconds.

  \item Distribute butter evenly and process until the mixture
    resembles coarse meal, 15~to 20~seconds.

  \item With the food processor running, pour the ice water in a
    steady stream through the feed tube and process just until the
    dough comes together.

  \end{enumerate}

  \ \vbox{
  \textbf{Manual Version}

  \begin{itemize}
    
  \item Whisk flour and salt in a large mixing bowl.

  \item Rub the butter into the flour with fingertips or cut it in
    with a pastry blender.

  \item Sprinkle the water over the mixture while tossing with a fork.

  \end{itemize}
  }

  \textbf{And then\dots}

  \begin{itemize}
    
  \item Knead the dough several turns on a lightly floured surface to
    bring it together.  Shape into one (or two) thick disks, wrap in
    plastic wrap, and refrigerate at least 1~hour.

  \item To roll out the dough, place the chilled dough between two
    pieces of baking parchment and roll out to a diameter two inches
    bigger than the size of the pan.  Fit to the pan, trim the edges,
    and keep in the refrigerator until ready to fill.  (If pre-baking
    the crust, refrigerate at least 30~minutes before baking.)

  \end{itemize}

\end{recipe}
