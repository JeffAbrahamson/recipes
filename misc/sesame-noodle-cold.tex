\dish{Cold Sesame Noodle}
\altdish{Sesame Noodle, Cold}
\altdish{Noodle, Cold Sesame}
\serves{3}
%\makes{}
\prep{15 minutes}
\source{internet}

\begin{ingredients}
  \ingrS{500}{g}{fresh noodles (or 230 g dried)}
  \ingr{40+}{g}{sesame paste (tahini) (3 \tbsp)}
  \ingr{75}{ml}{cold water (5 \tbsp)}
  \ingr{15}{ml}{light soy sauce (1 \tbsp)}
  \ingr{5--6}{g}{sugar (1 \tsp)}
  \ingr{5--6}{g}{salt (1 \tsp)}
  \ingr{7--8}{ml}{rice vinegar (or black vinegar) (\fracH \tbsp)}
  \ingr{10}{g}{sesame oil (1 \tbsp)}
  \ingrS{}{dash}{cayenne pepper}
  \ingr{}{}{crushed peanuts}
\end{ingredients}


\begin{recipe}
  \begin{enumerate}

  \item Mix sesame paste, water soy sauce, sugar, salt, vinegar, and
    sesame oil, whisking until smooth.  Add a dash of cayenne pepper.
    Set aside.
    
  \item Boil water and make noodles, draining and dousing in cold water when done.

  \item Toss noodles and sesame sauce together.  Serve topped with crushed peanuts.

  \end{enumerate}
\end{recipe}
