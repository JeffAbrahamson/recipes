\dish{Falafel}
\altdish{}
\serves{}
\makes{20 falafel balls}
\prep{1 hour}
\source{internet}

\stage{Falafel}

\begin{ingredients}
  \ingr{200}{g}{poids chiches (peser à sec), trempé dans de l'eau}
  \ingr{1}{}{oignon, haché}
  \ingr{2}{gousses}{d'ail, haché}
  \ingr{\fracH}{\cs}{sel}
  \ingr{\fracQ}{\ccf}{poivre noir moulu}
  \ingr{\fracH}{\cs}{coriandre en poudre}
  \ingr{1}{\ccf}{cumin en poudre}
  \ingr{}{}{coriandre feuille, haché}
\end{ingredients}

\begin{recipe}
  \begin{enumerate}

  \item Hacher les poids chiches 3~minutes en robo.

  \item Ajouter les autres ingrédients et hacher encore 3~minutes.

  \item Former des boules, légèrement aplaties afin d'avoir une cuisson plus homogène.

  \item Faire cuire 3~minutes dans l'huile de friture à 180\degreeC.

  \end{enumerate}
\end{recipe}

\stage{Sauce Tahini}

  \ingr{1}{gousse}{d'ail}

  \ingr{\fracH}{\cs}{sel}
  \ingr{3}{\cs}{eau}
\end{ingredients}


\begin{recipe}
  \begin{enumerate}

  \item Tout mélanger.

  \end{enumerate}
\end{recipe}

\begin{ingredients}
