\dish{Cinnamon Raisin Bread}
\altdish{Bread, cinnamon raisin}
%\serves{}
\makes{1 kg}
\prep{4.5 hours}
\source{modernist.bread}

\stage{Cinnamon Smear}

\begin{ingredients}
  \ingr{30}{g}{butter, softened}
  \ingr{25}{g}{light brown sugar}
  \ingr{10}{g}{pastry flour}
  \ingr{4}{g}{molasses}
  \ingr{3}{g}{cinnamon, ground}
  \ingr{1}{g}{vanilla}
  \ingr{0.1}{g}{salt}
\end{ingredients}


\begin{recipe}
  \begin{enumerate}

  \item Mix all ingredients with a rubber spatula until smooth.

  \end{enumerate}
\end{recipe}

\stage{Dough}

\begin{ingredients}
  \ingr{260}{g}{whole milk, cold}
  \ingr{10}{g}{instant dry yeast}
  \ingrS{2}{g}{malt syrup}
  \ingr{500}{g}{flour T65}
  \ingr{50}{g}{egg, cold (about one whole egg)}
  \ingr{45}{g}{butter, softened}
  \ingr{45}{g}{sugar}
  \ingr{10}{g}{salt}
\end{ingredients}

\begin{ingredients}
  \ingr{90}{g}{black raisins}
  \ingr{2}{g}{cinnamon, ground}
\end{ingredients}

Egg wash:
\begin{ingredients}
  \ingr{1}{}{whole egg}
  \ingr{1}{}{egg white}
\end{ingredients}


\begin{recipe}
  \begin{enumerate}

  \item Add yeast to milk and malt syrup.  Stir to dissolve.

  \item Add remaining ingredients except raisins and cinnamon.  Knead
    on medium speed to full gluten development.

  \item Cover and autolyse one hour with book fold after 30 minutes.

  \item Roll out dough to a rectangle the width of the baking pan,
    spread smear evenly, sprinkle with raisins and cinnamon.  Roll up,
    close seam, and place in pan, seam-side down.  Brush with egg wash.

  \item Proof 1.5 hours at 21\degreeC{} (or 1~hour at 27\degreeC).

  \item Preheat oven to 185\degreeC{} (convection) or 200\degreeC{} (conventional).

  \item Make three diagonal scores on the top.  Bake for
    25--30~minutes (convection) or 35--40 minutes (conventional).

  \item Cool 5~minutes in pan, then remove to wire rack and allow to
    cool completely.

  \end{enumerate}
\end{recipe}

\stage{Glazing}

\begin{ingredients}
  \ingr{20}{ml}{buttermilk}
  \ingrS{250}{g}{powdered sugar}
  \ingr{}{}{ground cinnamon}
\end{ingredients}

\begin{recipe}
  \begin{itemize}
  \item Mix buttermilk and sugar, apply in thin parallel lines on
    cooled bread, dust with cinnamon.
  \end{itemize}
\end{recipe}
