\dish{Cranberry Butter}
\altdish{Butter, cranberry}
\makes{1 cup}
\prep{15 minutes plus time to chill}
\source{ann:vc}

\begin{ingredients}
  \ingr{1}{c}{cranberry without liquid}
  \ingr{1 \fracH}{c}{confectioner's sugar}
  \ingr{\fracH}{c}{unsalted butter, softened and cut in pieces}
  \ingr{1}{Tbsp}{lemon juice}
\end{ingredients}


\begin{recipe}
  \begin{enumerate}

  \item Pur\'ee cranberry and sugar.  Add remaining ingredients.  Blend until smooth.

  \item Chill until firm.

  \end{enumerate}
\end{recipe}
