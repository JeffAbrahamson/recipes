\dish{かき揚げ}
\alsodish{Beignets de légumes croquants}
\serves{4}
%\makes{}
\prep{30 minutes}
\source{noriko-mitani}
\label{かき揚げ}

\stage{Beignets}

\begin{ingredients}
  \ingr{}{}{légumes coupées en julienne}
  \ingr{1}{c}{farine}
  \ingr{1}{\tbsp}{huile de sésame}
  \ingrS{\fracH}{c}{eau}
  \ingr{\fracH}{L}{huile de friture (tournesol, par exemple)}
\end{ingredients}

\begin{recipe}
  \begin{enumerate}

  \item Couper les légumes en julienne.
    
  \item Les mettre dans un grand bol, ajouter la farine et les graines
    de sésame, et mélanger le tout.
    
  \item Chauffer l'huile à 180\degreeC{} (approx) dans une poêle.  Avec
    l'aide de deux baguettes (ou deux fourchettes) faire frire par
    petites quantités.  Dorer légèrement les deux côtés jusqu'à ce que
    la pâte prenne une texture solide.
    
  \item Retirer les beignets de la poêle et mettre sur du papier
    absorbant.

  \end{enumerate}
\end{recipe}

\stage{Sauce}

\begin{ingredients}
  \ingr{\fracQ}{c}{shôyu}
  \ingr{\fracQ}{c}{mirin}
  \ingr{\fracQ}{c}{dashi}
\end{ingredients}

\begin{recipe}
  \begin{enumerate}
  \item Mélanger le tout ensemble.  Porter à ébullition.
  \item Servir tiède.
  \end{enumerate}
\end{recipe}
