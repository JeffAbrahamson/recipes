\dish{Bao}
\altdish{Steamed buns}
%\serves{}
\makes{12}
\prep{3 hours}
\source{modernist.bread}

\begin{ingredients}
  \ingr{305}{g}{water}
  \ingr{45}{g}{nonfat milk powder}
  \ingrS{9}{g}{dry yeast}
  \ingr{470}{g}{bread flour}
  \ingr{85}{g}{sugar}
  \ingr{55}{g}{potato starch}
  \ingr{20}{g}{vegetable oil}
  \ingr{10}{g}{salt}
\end{ingredients}


\begin{recipe}
  \begin{enumerate}

  \item Combine water, milk powder and yeast.  Stir to dissolve.

  \item Add remaining ingredients and mix on low speed to a shaggy
    mass.  Mix on medium speed to full gluten development.

  \item Transfer to an oiled tub and cover.  Bulk ferment 45 minutes,
    no folds.

  \item Divide into 12 balls of 70~g each.  Preform as boules.

  \item Prepare 12~squares of parchment paper and brush lightly with
    oil.

  \item Shape a round bun, then flatten to a disc, rolling the edges
    thinner than the middle.  Place 35~g of filling at the center,
    bring the edges to a close around the filling.  Place each bun on
    an oiled piece of parchment paper.

  \item Final proof, 30--40~minutes.

  \item Steam, seam-side down, on the parchment paper.

  \item Consume within 10~minutes while the buns are still hot.  They
    become substantially less interesting once they cool.

  \end{enumerate}
\end{recipe}
