\dish{Polenta}
%\altdish{}
%\serves{}
%\makes{}
\prep{15 minutes}
\source{stephane.birkle}

\begin{ingredients}
  \ingr{125}{g}{polenta (semoule de maïs)}
  \ingr{500}{ml}{eau}
  \ingr{50}{g}{beurre}
  \ingr{50}{g}{parmesan rapé}
\end{ingredients}


\begin{recipe}
  \begin{enumerate}

  \item Faire fondre le beurre dans l'eau chaude, puis ajouter le
    semoule de maïs.

  \item Vers la fin de la cuisson, ajouter le parmesan rapé.

  \item Servir chaud.

  \end{enumerate}
\end{recipe}
