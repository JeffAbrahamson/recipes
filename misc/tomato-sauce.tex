\dish{Tomato sauce}
\altdish{Sauce, tomato}
%\serves{}
%\makes{}
\prep{5 minutes + time to drain}
\source{modernist.bread}
\label{tomato.sauce}

\begin{ingredients}
  \ingr{360}{g}{canned tomato, drained}
  \ingr{125}{g}{tomato paste}
  \ingr{0.5}{g}{oregano, dried}
  \ingr{5}{g}{salt}
  \ingr{7}{g}{extra-virgin olive oil}
\end{ingredients}


\begin{recipe}
  \begin{enumerate}

  \item Drain tomatoes, then combine all ingredients, optionally
    blending with an immersion blender.

  \item Store in an air-tight container up to 1~week refrigerated or
    3~months frozen.

  \end{enumerate}
\end{recipe}
