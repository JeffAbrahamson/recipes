\dish{Dill Pickle Slices}
\altdish{Pickles, Dill Slices}
%\serves{}
%\makes{}
\prep{15 minutes}
\source{jma}

\begin{ingredients}
  \ingr{1}{c}{white wine vinegar}
  \ingr{\fracH}{c}{sugar}
  \ingr{1}{Tbsp}{fresh dill, chopped}
  \ingr{}{}{water}
  \ingr{3}{}{large cucumbers}
  \ingr{6}{}{cloves garlic}
\end{ingredients}


\begin{recipe}
  \begin{enumerate}

  \item Mix the vinegar and sugar until dissolved.  Place in jars of
    sufficient size to hold sliced cucumbers.

  \item Slice cucumbers and garlic.  Add alternately to jars along
    with dill.  Add enough water to cover.

  \item Refrigerate for two days before eating.

  \end{enumerate}
\end{recipe}
