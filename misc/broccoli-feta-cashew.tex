\dish{Broccoli pur\'ee with feta and cashew}
\altdish{Feta and cashew pur\'ee with broccoli}
\altdish{Cashew and feta pur\'ee with broccoli}
\altdish{Pur\'ee of broccoli, cashew, and feta}
\serves{4--8}
%\makes{}
\prep{}
\source{unknown}

\begin{ingredients}
  \ingr{500}{g}{onion}
  \ingr{6}{cloves}{garlic}
  \ingr{130}{g}{cashews, roasted, unsalted}
  \ingr{2}{}{carrots}
  \ingr{}{}{olive oil}
  \ingr{3}{}{head broccoli}
  \ingr{}{}{thyme}
  \ingr{}{}{red pepper or cayenne pepper}
  \ingr{100}{g}{feta}
\end{ingredients}


\begin{recipe}
  \begin{enumerate}

  \item Saute onions until soft.

  \item Add carrots and broccoli.  Saute until the broccoli is
    slightly soft and dark green.

  \item Add spices.  Pur\'ee with feta.

  \end{enumerate}
\end{recipe}
