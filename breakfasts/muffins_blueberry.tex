\dish{Blueberry Muffins}
\altdish{Muffins, blueberry}
%\serves{}
\makes{16 muffins}
\prep{30 minutes}
\source{stephane.birkle}

\begin{ingredients}
  \ingr{360}{g}{plain flour}
  \ingr{370}{g}{caster sugar}
  \ingr{1 \fracH}{\tsp}{baking powder}
  \ingr{1}{\tsp}{baking soda}
  \ingr{375}{ml}{buttermilk}
  \ingr{1}{}{egg}
  \ingr{\fracH}{\tsp}{vanilla extract}
  \ingr{70}{g}{unsalted butter}
  \ingr{250}{g}{blueberries}
\end{ingredients}


\begin{recipe}
  \begin{enumerate}

  \item Preheat oven to 180\C.

  \item Mix dry ingredients and wet ingredients separately.  Combine.
  Bake in muffin tins.

  \item Bake for 20--25 minutes.

  \end{enumerate}
\end{recipe}
